<head>
  <title>Aba Interativa</title>
  <style>
    /* Estilos para as abas */
    .tab {
      overflow: hidden;
      border: 1px solid #ccc;
      background-color: #f1f1f1;
    }

    /* Estilos para os botões de abas */
    .tab button {
      background-color: inherit;
      float: left;
      border: none;
      outline: none;
      cursor: pointer;
      padding: 14px 16px;
      transition: 0.3s;
    }

    /* Estilo para o conteúdo das abas */
    .tabcontent {
      display: none;
      padding: 6px 12px;
      border: 1px solid #ccc;
      border-top: none;
    }
  </style>
</head>
<body>
  <!-- Elemento de abas -->
  <div class="tab">
    <!-- Botões de abas -->
    <button class="tablinks" onclick="openTab(event, 'Tab1')">Aba 1</button>
    <button class="tablinks" onclick="openTab(event, 'Tab2')">Aba 2</button>
    <button class="tablinks" onclick="openTab(event, 'Tab3')">Aba 3</button>
  </div>

  <!-- Conteúdo das abas -->
  <div id="Tab1" class="tabcontent">
    <h3>Conteúdo da Aba 1</h3>
    <p>Este é o conteúdo da aba 1.</p>
  </div>

  <div id="Tab2" class="tabcontent">
    <h3>Conteúdo da Aba 2</h3>
    <p>Este é o conteúdo da aba 2.</p>
  </div>

  <div id="Tab3" class="tabcontent">
    <h3>Conteúdo da Aba 3</h3>
    <p>Este é o conteúdo da aba 3.</p>
  </div>

  <script>
    // Função para alternar entre as abas
    function openTab(evt, tabName) {
      // Oculta todos os elementos com a classe "tabcontent"
      var tabcontent = document.getElementsByClassName("tabcontent");
      for (var i = 0; i < tabcontent.length; i++) {
        tabcontent[i].style.display = "none";
      }

      // Remove a classe "active" de todos os botões de abas
      var tablinks = document.getElementsByClassName("tablinks");
      for (var i = 0; i < tablinks.length; i++) {
        tablinks[i].className = tablinks[i].className.replace(" active", "");
      }

      // Exibe o conteúdo da aba selecionada e adiciona a classe "active" ao botão da aba
      document.getElementById(tabName).style.display = "block";
      evt.currentTarget.className += " active";
    }
  </script>
